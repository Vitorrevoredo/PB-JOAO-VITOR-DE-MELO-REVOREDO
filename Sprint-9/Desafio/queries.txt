1. Quais são os artistas com maior número de participações em obras, agrupados por gênero?

SELECT 
    da.nomeartista,
    dg.nome_genero,
    COUNT(*) AS num_obras
FROM 
    fatoobras fo
JOIN 
    dimartista da ON fo.artista_id = da.artista_id
JOIN 
    dimobra do ON fo.obra_id = do.obra_id
JOIN 
    dimgenero dg ON do.genero_id = dg.genero_id
GROUP BY 
    da.nomeartista, dg.nome_genero
ORDER BY 
    num_obras DESC;

2. Qual o tempo médio de duração dos filmes em cada gênero?

SELECT 
    dg.nome_genero,
    AVG(CASE 
            WHEN fo.tempominutos = '\N' THEN NULL  
            ELSE CAST(fo.tempominutos AS DOUBLE)   
        END) AS avg_duration
FROM 
    fatoobras fo
JOIN 
    dimobra do ON fo.obra_id = do.obra_id
JOIN 
    dimgenero dg ON do.genero_id = dg.genero_id
WHERE 
    fo.tempominutos IS NOT NULL AND fo.tempominutos <> '\N'
GROUP BY 
    dg.nome_genero
ORDER BY 
    avg_duration DESC;
    
3. Quais filmes de Comédia e Animação têm as melhores avaliações?

WITH ObraGeneroRankeada AS (
    SELECT
        do.titulopincipal,
        dg.nome_genero,
        AVG(fo.vote_average) AS avg_vote,
        RANK() OVER (
            PARTITION BY do.titulopincipal
            ORDER BY AVG(fo.vote_average) DESC, dg.nome_genero ASC
        ) AS ranking
    FROM 
        fatoobras fo
    JOIN 
        dimobra do ON fo.obra_id = do.obra_id
    JOIN 
        dimgenero dg ON do.genero_id = dg.genero_id
    WHERE 
        dg.nome_genero IN ('Comedy', 'Animation')
    GROUP BY 
        do.titulopincipal, dg.nome_genero
)
SELECT 
    titulopincipal,
    nome_genero,
    avg_vote
FROM 
    ObraGeneroRankeada
WHERE 
    ranking = 1
ORDER BY 
    avg_vote DESC;

4. Quantos títulos foram lançados em cada ano?

SELECT 
    dt.ano AS year,
    COUNT(*) AS num_titulos
FROM 
    fatoobras fo
JOIN 
    dimtempo dt ON fo.tempo_id = dt.tempo_id
GROUP BY 
    dt.ano
ORDER BY 
    year ASC;

5. Qual é a relação entre o orçamento e a receita para cada gênero?

SELECT 
    dg.nome_genero,
    AVG(do.budget) AS avg_budget,
    AVG(do.revenue) AS avg_revenue,
    AVG(do.revenue / NULLIF(do.budget, 0)) AS retorno_investimento
FROM 
    dimobra do
JOIN 
    dimgenero dg ON do.genero_id = dg.genero_id
GROUP BY 
    dg.nome_genero
ORDER BY 
    retorno_investimento DESC;

7. Quais filmes de animação têm o maior orçamento?

SELECT 
    do.titulopincipal,
    do.budget
FROM 
    dimobra do
JOIN 
    dimgenero dg ON do.genero_id = dg.genero_id
WHERE 
    dg.nome_genero = 'Animation'

    do.budget DESC

8. Quais filmes tiveram o maior lucro (receita - orçamento) no gênero "Animação"?

SELECT 
    do.titulopincipal,
    (do.revenue - do.budget) AS lucro
FROM 
    dimobra do
JOIN 
    dimgenero dg ON do.genero_id = dg.genero_id
WHERE 
    dg.nome_genero = 'Animation'
ORDER BY 
    lucro DESC
LIMIT 10;

9. Quais anos foram mais significativos em termos de lançamento e avaliação?

SELECT 
    dt.ano AS year,
    COUNT(*) AS num_lancamentos,
    AVG(fo.vote_average) AS avg_vote
FROM 
    fatoobras fo
JOIN 
    dimtempo dt ON fo.tempo_id = dt.tempo_id
WHERE 
    fo.vote_average IS NOT NULL
GROUP BY 
    dt.ano
ORDER BY 
    num_lancamentos DESC, avg_vote DESC;

11. Qual o retorno financeiro médio dos filmes com maior sucesso (considerando a diferença entre revenue e budget)?

SELECT 
    do.titulopincipal,
    AVG(CASE WHEN fo.revenue IS NULL OR fo.budget = 0 THEN NULL
        ELSE (fo.revenue - fo.budget) / fo.budget END) AS avg_return
FROM 
    fatoobras fo
JOIN 
    dimobra do ON fo.obra_id = do.obra_id
GROUP BY 
    do.titulopincipal
ORDER BY 
    avg_return DESC;

12. Quais artistas têm as obras com maior lucro?

SELECT 
    da.nomeartista,
    (SUM(do.revenue) - SUM(do.budget)) AS lucro_total
FROM 
    fatoobras fo
JOIN 
    dimartista da ON fo.artista_id = da.artista_id
JOIN 
    dimobra do ON fo.obra_id = do.obra_id
GROUP BY 
    da.nomeartista
ORDER BY 
    lucro_total DESC;
13. Qual foi o filme de animação mais lucrativo em relação ao orçamento?

SELECT 
    do.titulopincipal,
    (do.revenue - do.budget) / NULLIF(do.budget, 0) AS retorno_investimento
FROM 
    dimobra do
JOIN 
    dimgenero dg ON do.genero_id = dg.genero_id
WHERE 
    dg.nome_genero = 'Animation'
ORDER BY 
    retorno_investimento DESC
LIMIT 1;

14. Quais são os filmes de maior orçamento que não atingiram o sucesso financeiro esperado?

SELECT 
    do.titulopincipal,
    do.budget,
    do.revenue,
    (do.revenue - do.budget) AS lucro
FROM 
    dimobra do
JOIN 
    dimgenero dg ON do.genero_id = dg.genero_id
WHERE 
    do.budget > 1000000
    AND do.revenue < do.budget
ORDER BY 
    lucro ASC
LIMIT 10;

15. Quais gêneros de filmes tiveram os maiores lançamentos globais (por receita)?

SELECT 
    dg.nome_genero,
    SUM(do.revenue) AS total_revenue
FROM 
    dimobra do
JOIN 
    dimgenero dg ON do.genero_id = dg.genero_id
WHERE 
    do.revenue > 0
GROUP BY 
    dg.nome_genero
ORDER BY 
    total_revenue DESC;
